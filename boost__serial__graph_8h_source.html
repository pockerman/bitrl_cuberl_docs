287std::vector<uint_t>
288BoostSerialGraph<VertexData,EdgeData>::get_vertex_neighbors_ids(uint_t id)const{
289
290 if(id >=n_vertices()){
291 throw std::logic_error("Invalid vertex index. Index "+
292 std::to_string(id)+
293 " not in [0,"+
294 std::to_string(n_vertices())+
295 ")");
296 }
297
298
299 auto vneighs = get_vertex_neighbors(id);
300 std::vector<uint_t> neighbors;
301 neighbors.reserve(std::distance(vneighs.first, vneighs.second));
302
303 auto start = vneighs.first;
304 auto end = vneighs.second;
305 while(start != end){
306 auto& vertex = get_vertex(start);
307 neighbors.push_back(vertex.id);
308 ++start;
309 }
310
311 neighbors;
312}

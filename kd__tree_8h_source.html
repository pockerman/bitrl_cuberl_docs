104template<typename Node, typename DataType>
105int
106compare(std::shared_ptr<Node> node, const DataType& data, const uint_t k){
107
108 auto node_key = node->get_point_key(node->level, k, node->data);
109 auto point_key = node->get_point_key(node->level, k, data);
110
111 auto sign = point_key - node_key;
112
113 if (sign < 0){
114 return -1;
115 }
116 else if(sign > 0){
117 return 1;
118 }
119
125 return node->level % 2 == 0 ? -1 : 1;
126}

150template<typename Iterator, typename ComparisonPolicy>
151std::tuple<typename std::iterator_traits<Iterator>::value_type,
152 std::pair<Iterator, Iterator>,
153 std::pair<Iterator, Iterator>>
154partiion_on_median(Iterator begin, Iterator end, uint_t level, uint_t k, const ComparisonPolicy& comp_policy){
155
156
157 typedef typename std::iterator_traits<Iterator>::value_type value_type;
158
159 auto begin_ = begin;
160 auto end_ = end;
161
162 auto n_points = std::distance(begin_, end_);
163
164 // the median index
165 auto median_idx = n_points % 2 == 0 ? (n_points + 1) / 2 : n_points / 2;
166
167 auto compare = [&](const value_type& v1, const value_type& v2){
168
169 auto idx = level % k;
170 return comp_policy(v1, v2, idx); //v1[idx] < v2[idx];
171 };
172
173
174 // rearrange the elements
175 std::nth_element(begin, begin + median_idx, end , compare);
176
177 auto median = *(begin + median_idx);
178
179 auto left = std::make_pair<Iterator, Iterator>(std::forward<Iterator>(begin), begin + median_idx);
180 auto right = std::make_pair<Iterator, Iterator>(begin + median_idx + 1, std::forward<Iterator>(end));
181 return std::make_tuple(median, left, right);
182
183}

376std::vector<std::pair<typename ComparisonPolicy::value_type, typename NodeType::data_type>>
377KDTree<NodeType>::nearest_search_(std::shared_ptr<node_type> node, const data_type& data,
378 uint_t n, const ComparisonPolicy& calculator)const{
379
380 typedef std::pair<typename ComparisonPolicy::value_type, std::shared_ptr<node_type>> pair_value_type;
381
382 struct comparison
383 {
384 //typedef std::greater<typename ComparisonPolicy::value_type> compare_op;
385 bool operator()(const pair_value_type& v1, const pair_value_type& v2)const{
386 return v1.first > v2.first;
387 }
388 };
389
390 // initialize a min-heap
391 //std::priority_queue<value_type, decltype(compare)> pq;
392 cubeai::containers::FixedSizeMinPriorityQueue<pair_value_type, comparison> pq(n);
393
400 pq.push(std::make_pair(std::numeric_limits<typename ComparisonPolicy::value_type>::max(),std::shared_ptr<node_type>()));
401
402 do_nearest_search_(node, data, calculator, pq);
403
404 auto peek = pq.top();
405
410
411 if(!peek.second){
412 pq.pop();
413 }
414
415 std::vector<std::pair<typename ComparisonPolicy::value_type, typename NodeType::data_type>> result;
416 result.reserve(pq.size());
417 while(!pq.empty()){
418 auto item = pq.top_and_pop();
419 result.push_back({item.first, item.second->data});
420 }
421
422 // loop over the priority queue and collect
423 // the points we found

425 return result;

430void
431KDTree<NodeType>::do_nearest_search_(std::shared_ptr<node_type> node, const data_type& data,
432 const ComparisonPolicy& calculator, PriorityQueueType& pq)const{
433
434
435 if(!node){
436 // nothing to do with the queue
437 return;
438 }
439 else{
440
441 // compute the distance between the target
442 // and the data hold by the node
443 auto dist = calculator.evaluate(node->data, data);
444
445 // insrt into the queue
446 pq.push(std::make_pair(dist, node));
447
448 if(detail::compare(node, data, k_) < 0){
449
450 auto close_branch = node->left;
451 do_nearest_search_(close_branch, data, calculator, pq );
452
453 auto peek = pq.top();
454
455 if(detail::split_distance(node, data, k_) < peek.first){
456 do_nearest_search_(node->right, data, calculator, pq );
457 }
458 }
459 else{
460
461 auto close_branch = node->right;
462 do_nearest_search_(close_branch, data, calculator, pq );
463
464 auto peek = pq.top();
465
466 if(cubeai::containers::detail::split_distance(node, data, k_) < peek.first){
467 do_nearest_search_(node->left, data, calculator, pq );
468 }
469 }
470 }

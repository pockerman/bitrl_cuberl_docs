202template<typename ObstacleTp>
203real_t
204DiffDriveDW::calc_obstacle_cost_(const trajectory_t& trajectory, const ObstacleTp& obstacle){
205
206 // calc obstacle cost max (inf): collistion, 0:free
207 auto minr = std::numeric_limits<real_t>::max();
208
209 for (auto ii=0; ii<trajectory.size(); ii += this->config_.skip_n){
210 for (auto i=0; i< obstacle.size(); i++){
211
212 auto ox = obstacle[i][0];
213 auto oy = obstacle[i][1];
214 auto dx = trajectory[ii][0] - ox;
215 auto dy = trajectory[ii][1] - oy;
216
217 float r = std::sqrt(dx*dx + dy*dy);
218
219 if (r <= this->config_.robot_radius){
220 return std::numeric_limits<real_t>::max();
221 }
222
223 if (minr >= r){
224 minr = r;
225 }
226 }
227 }
228
229 return 1.0 / minr;
230}

234template<typename ObstacleTp>
235DiffDriveDW::trajectory_t
236DiffDriveDW::calculate_control_input_(const ObstacleTp& obstacle){
237
238 auto min_cost = std::numeric_limits<real_t>::max(); //this->config_.min_cost;
239 auto min_u = control_;
240 min_u[0] = 0.0;
241 min_u[1] = 0.0;
242
243 auto v_min = w_properties_.v_min;
244 auto v_max = w_properties_.v_max;
245 auto w_min = w_properties_.w_min;
246 auto w_max = w_properties_.w_max;
247
248 // the best trajectory
249 trajectory_t best_traj;
250
251 // evalucate all trajectory with sampled input in dynamic window
252 for (auto v=v_min; v <= v_max; v += this->config_.v_reso){
253 for (auto w=w_min; w <= w_max; w += this->config_.yawrate_reso){
254
255
256
257 trajectory_t traj = predict_trajectory_(v, w);
258
259 // what's the cost reaching goal on this
260 // trajectory
261 auto to_goal_cost = this->config_.to_goal_cost_gain*calc_to_goal_cost_(traj);
262 auto speed_cost = this->config_.speed_cost_gain * (this->config_.max_speed - traj.back()[3]);
263 auto ob_cost = this->config_.obstacle_cost_gain*calc_obstacle_cost_(traj, obstacle);
264
265 // final cost
266 auto final_cost = to_goal_cost + speed_cost + ob_cost;
267
268 // we want minimum cost trajectory
269 if (min_cost >= final_cost){
270 min_cost = final_cost;
271 min_u[0] = v;
272 min_u[1] = w; //control_t{{v, w}};
273 best_traj = traj;
274
275 if(std::fabs(min_u[0]) < config_.robot_stuck_flag_cons &&
276 std::fabs(state_["v"]) < config_.robot_stuck_flag_cons){
277 // to ensure the robot do not get stuck in
278 // best v=0 m/s (in front of an obstacle) and
279 // best omega=0 rad/s (heading to the goal with
280 // angle difference of 0)
281 min_u[1] = -config_.max_delta_yaw_rate;
282 }
283 }
284 }
285 }
286
287 // update the control and return the best
288 // trajectory
289 control_ = min_u;
290 return best_traj;
291}

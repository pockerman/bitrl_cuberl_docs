65 typename LQRController<StateType, StateSize, InputSize, T>::ctrl_signal_type
66 LQRController<StateType, StateSize, InputSize, T>::compute_ctrl_signal(const StateType& init, const StateType& target, real_t dt)
67 {
68 typedef StateType state_type;
69 typedef typename LQRController<StateType, StateSize, InputSize, T>::ctrl_signal_type result;
70 result path{initial};
71 path.reserve(static_cast<uint_t>(std::round(max_time / dt + 1))); // TODO: currently assuming the worst case
72
73 state_type x = initial - target;
74 VectorMx1 u;
75
76 bool path_found = false;
77 double time = 0;
78 double goal_dist_squared = std::pow(goal_dist, 2);
79
80 while (time < max_time) {
81 time += dt;
82
83 u = Input(x);
84 x = A * x + B * u;
85
86 path.push_back(x + target);
87
88 if (x.squaredNorm() <= goal_dist_squared) {
89 path_found = true;
90 break;
91 }
92 }
93
94 if (!path_found) {
95 std::cerr << "Couldn't find a path\n";
96 return {};
97 }
98
99 return path;
100 }

102 template<typename StateType, uint_t StateSize, uint_t InputSize, typename T>
103 SquareMat<T, StateSize>
104 LQRController<StateType, StateSize, InputSize, T>::solve_lqr()
105 {
106 SquareMat<T, StateSize> X = sys_description_.Q;
107 SquareMat<T, StateSize> Xn = sys_description_.Q;
108
109 const auto& A = sys_description_.A;
110 const auto& B = sys_description_.B;
111 const auto& R = sys_description_.R;
112 const auto& Q = sys_description_.Q;
113 for (auto _ = 0; _ < max_iter_; _++) {
114 Xn = A.transpose() * X * A
115 - A.transpose() * X * B * (R + B.transpose() * X * B).inverse() * B.transpose()
116 * X * A
117 + Q;
118 if ((Xn - X).template lpNorm<Eigen::Infinity>() < eps_) break;
119 X = Xn;
120 }
121 return Xn;
122 }
